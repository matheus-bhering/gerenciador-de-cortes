Cole essa URL no arquivo index.html (variável APPS_SCRIPT_URL) // ===================================================== const SHEET_NAME = "dashboard"; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter; const action = params.action; let result; try { switch (action) { case "read": result = readAll(); break; case "add": result = addRow(params); break; case "updatePayment": result = updatePayment(params); break; case "updateInvoice": result = updateInvoice(params); break; case "deleteRow": result = deleteRow(params); break; default: result = { success: false, error: "Ação inválida" }; } } catch (err) { result = { success: false, error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function readAll() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = []; for (let i = 1; i < data.length; i++) { const row = {}; for (let j = 0; j < headers.length; j++) { let val = data[i][j]; // Format dates if (val instanceof Date) { val = Utilities.formatDate(val, Session.getScriptTimeZone(), "dd/MM/yyyy"); } row[headers[j]] = val; } rows.push(row); } return { success: true, data: rows }; } function addRow(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); // Generate next ID let maxId = 0; for (let i = 1; i < data.length; i++) { const id = parseInt(data[i][0]); if (!isNaN(id) && id > maxId) maxId = id; } const newId = maxId + 1; const newRow = [ newId, params.data || "", params.tipoCorte || "", params.preco ? parseFloat(params.preco.replace(",", ".")) : "", params.status || "Pendente", params.dataPagamento || "", params.notaFiscal || "", params.observacoes || "" ]; sheet.appendRow(newRow); return { success: true, message: "Corte adicionado!", id: newId }; } function updatePayment(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const targetId = parseInt(params.id); for (let i = 1; i < data.length; i++) { if (parseInt(data[i][0]) === targetId) { sheet.getRange(i + 1, 5).setValue("Pago"); // Status sheet.getRange(i + 1, 6).setValue(params.dataPagamento || Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "dd/MM/yyyy")); // Data Pagamento return { success: true, message: "Pagamento confirmado!" }; } } return { success: false, error: "ID não encontrado" }; } function updateInvoice(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const targetId = parseInt(params.id); for (let i = 1; i < data.length; i++) { if (parseInt(data[i][0]) === targetId) { sheet.getRange(i + 1, 7).setValue(params.notaFiscal || ""); // Nota Fiscal return { success: true, message: "Nota fiscal atualizada!" }; } } return { success: false, error: "ID não encontrado" }; } function deleteRow(params) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const targetId = parseInt(params.id); for (let i = 1; i < data.length; i++) { if (parseInt(data[i][0]) === targetId) { sheet.deleteRow(i + 1); return { success: true, message: "Registro deletado!" }; } } return { success: false, error: "ID não encontrado" }; }
